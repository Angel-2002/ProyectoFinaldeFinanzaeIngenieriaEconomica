# ProyectoFinaldeFinanzaeIngenieriaEconomica
Python (FastAPI) and SQL

# Crear entorno dentro de la carpeta
py -m venv env

# Instalar
- pip install fastapi
- pip install uvicorn
- pip install sqlalchemy
- pip install pymysql
- pip install bcrypt

# Activar entorno
env/Scripts/activate

# Comando para ejecutar localmente
uvicorn main:app --reload
